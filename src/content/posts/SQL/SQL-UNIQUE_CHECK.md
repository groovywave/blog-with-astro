**CHECK**

DROP TABLE IF EXISTS posts;
CREATE TABLE posts (
message VARCHAR(140) UNIQUE,
like INT CHECK (likes >= 0 AND likes <= 100)
);

INSERT INTO posts (message, likes) VALUES
('Thanks', 12),
('Arigato', 4),
('Merci', 4);

INSERT INTO posts (message) VALUES ('Gracias');

SELECT * FROM posts;
